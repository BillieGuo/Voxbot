# Query: Increment_rpy r='0.0', p='-0.1', y='0.0'
commands = [
    'source ~/interbotix_ws/install/setup.bash',
    '''ros2 topic pub /master_control/command widowx_msg/msg/WidowXCommand '{
        "cmd_type": "Increment rpy",
        "rpy": [0.0, -0.1, 0.0]
    }' --once'''.toString()
]
for command in commands:
    result = subprocess.run(command, shell=True, executable='/bin/bash', capture_output=True, text=True)
    print(result.stdout)
    print(result.stderr)
# done
